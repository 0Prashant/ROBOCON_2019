20 int8_t MPU6050_Init(struct MPU6050 *mpu)

24  HAL_StatusTypeDef h1 = HAL_I2C_Master_Transmit(mpu->hi2c, mpu->address, i2c_buffer, 2, 10);

32  HAL_I2C_Master_Transmit(mpu->hi2c, mpu->address, i2c_buffer, 2, 10);
33 
34  i2c_buffer[0] = GYRO_CONFIG;
35  i2c_buffer[1] = (mpu->g_scale) << 3;
36  HAL_StatusTypeDef h3 = HAL_I2C_Master_Transmit(mpu->hi2c, mpu->address, i2c_buffer, 2, I2C_TIMEOUT);
37 
38  i2c_buffer[0] = ACCEL_CONFIG;
39  i2c_buffer[1] = (mpu->a_scale) << 3;
40  HAL_StatusTypeDef h4 = HAL_I2C_Master_Transmit(mpu->hi2c, mpu->address, i2c_buffer, 2, I2C_TIMEOUT);
41 
42  return (h1 | h2 | h3 | h4);
43 }

140 Vec3<float> MPU6050_Calc_OmegaBias(struct MPU6050 *mpu, uint32_t n)
141 {
142  Vec3<float> bias;
143  Vec3<float> gyro;
144  float item_num;
145 
146  Exp_Smooth x_alpha35(0.35);
147  Exp_Smooth y_alpha35(0.35);
148  Exp_Smooth z_alpha35(0.35);
149 
150  float gx,gy,gz;
151 
152  for (uint32_t i = 0; i < n; ++i) {
153  MPU6050_Read_NormGyro(mpu);
154  gyro = mpu->norm_g_axis;
155 
156  gx = x_alpha35.smooth(gyro.getX());
157  gy = y_alpha35.smooth(gyro.getY());
158  gz = z_alpha35.smooth(gyro.getZ());
159 
160  gyro.set_Values(gx, gy, gz);
161 
162  item_num = 1.0 / (float)(i+1);
163  bias += (gyro - bias).mult_EW(item_num);
164 
165  HAL_Delay(1);
166  }
167  return bias;
